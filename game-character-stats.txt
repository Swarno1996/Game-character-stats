class GameCharacter:
    def __init__(self, name):
        self._name = name
        self._health = 100
        self._mana = 50
        self._level = 1
    
    @property
    #name property for read only access
    def name(self):
        return self._name
    
    #health property
    @property
    def health(self):
        return self._health

    @health.setter
    def health(self, new_health):
        if new_health < 0:
            self._health = 0
        elif new_health > 100:
            self._health = 100
        else:
            self._health = new_health
    
    #mana property
    @property
    def mana(self):
        return self._mana
    
    @mana.setter
    def mana(self, new_mana):
        if new_mana < 0:
            self._mana = 0
        elif new_mana > 50:
            self._mana = 50
        else:
            self._mana = new_mana
    
    #level getter
    @property
    def level(self):
        return self._level
    
    # level_up method
    def level_up(self):
            self._level += 1
            self.health = 100
            self.mana = 50
            print(f"{self._name} leveled up to {self._level}!")

    #str method
    def __str__(self):
        return(
        f"Name: {self.name}\n"
        f"Level: {self.level}\n"
        f"Health: {self.health}\n"
        f"Mana: {self.mana}")
